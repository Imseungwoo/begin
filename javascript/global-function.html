setTimeout 과 setInterval

function test(){
    console.log("test()");
}

var id = setTimeout(test, 1000); //1초후 1번실행

var id = setInterval(test, 1000); //1초마다 계속 실행

//setInterval 보다 나은 방법
function test(){
    console.log("test()");
    setTimeout(test, 1000);
}
